Process a natural language message and return an intelligent response with coordinated actions.

## Headers


| Name | Required | Description |
| --- | --- | --- |
| Authorization | Yes | AWS4-HMAC-SHA256 signature |
| X-Amz-Content-Sha256 | Yes | SHA256 hash of the request body |
| X-Amz-Date | Yes | ISO8601 timestamp |

(function(){
  const form = document.getElementById('f-post_roomi');

  if(!form||!out) return;
  resetBtn && resetBtn.addEventListener('click', ()=>{ form.reset(); out.textContent=''; });
  form.addEventListener('submit', async (e)=>{
    e.preventDefault();
    try {
      const body = {};
{ const v = document.getElementById("fld_user_id").value; if(v!=="" || true) body["user_id"] = (String)(v); }
{ const v = document.getElementById("fld_unit_id").value; if(v!=="" || true) body["unit_id"] = (String)(v); }
{ const v = document.getElementById("fld_unit_name").value; if(v!=="" || true) body["unit_name"] = (String)(v); }
{ const v = document.getElementById("fld_transcript").value; if(v!=="" || true) body["transcript"] = (String)(v); }

      const res = await fetch('https://ai-{env}-ae.roomi-services.com/v1/roomi', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body)
