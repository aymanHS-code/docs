Create a new log entry in CloudWatch.

## Headers


| Name | Required | Description |
| --- | --- | --- |
| Authorization | Yes | AWS4-HMAC-SHA256 signature |
| X-Amz-Content-Sha256 | Yes | SHA256 hash of the request body |
| X-Amz-Date | Yes | ISO8601 timestamp |



<RequestExample>

```json Request Body
{
  "message": "message",
  "group": "group",
  "stream": "stream",
  "metadata": {}
}
```

</RequestExample>


<ResponseExample>

```json Response
{
  "status": "status",
  "response": {}
}
```

</ResponseExample>

(function(){

    e.preventDefault();
    try {
      const body = {};
{ const v = document.getElementById("fld_message").value; if(v!=="" || true) body["message"] = (String)(v); }
{ const v = document.getElementById("fld_group").value; if(v!=="" || true) body["group"] = (String)(v); }

      const res = await fetch('https://logger-{env}-ae.roomi-services.com/logs', {
