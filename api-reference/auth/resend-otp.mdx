Resend OTP using pin ID.

## Headers


| Name | Required | Description |
| --- | --- | --- |
| Authorization | Yes | AWS4-HMAC-SHA256 signature |
| X-Amz-Content-Sha256 | Yes | SHA256 hash of the request body |
| X-Amz-Date | Yes | ISO8601 timestamp |
| X-Unit-ID | Yes | Unit identifier |
| X-User-ID | Yes | User identifier |


## Body

```json
{
  "pin": "pin-123"
}
```

(function(){
  const form = document.getElementById('f-post_api_resend-otp');
  const out = document.getElementById('resp-post_api_resend-otp');
  const resetBtn = document.getElementById('reset-post_api_resend-otp');
  if(!form||!out) return;
  resetBtn && resetBtn.addEventListener('click', ()=>{ form.reset(); out.textContent=''; });
  form.addEventListener('submit', async (e)=>{
    e.preventDefault();
    try {
      const body = {};
{ const v = document.getElementById("fld_pin").value; if(v!=="" || false) body["pin"] = (String)(v); }
      const res = await fetch('https://auth-{env}-ae.roomi-services.com/api/resend-otp', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body)
      });
      const text = await res.text();
      try { out.textContent = JSON.stringify(JSON.parse(text), null, 2); } catch { out.textContent = text; }
    } catch(err) {
      out.textContent = String(err);
    }
  });
})();
